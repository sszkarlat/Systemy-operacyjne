#!/usr/bin/env bash

var="pkn"
arr=("$@")
wygrane=0
porażki=0

for (( i=1; i<="$#"; i++ )) do
    comp="${var:$(( RANDOM % ${#var} )):1}"
    case ${arr[$i]} in
    "p")    case $comp in
            "k") (( wygrane++ ))
                 echo "$1: ${arr[$i]} - $comp";;
            "n") (( porażki++ ))
                 echo "$1: ${arr[$i]} - $comp" ;;
            *) echo "$1: ${arr[$i]} - $comp" ;; 
        esac
        ;;
    "k")    case $comp in
            "p") (( porażki++ ))
                 echo "$1: ${arr[$i]} - $comp" ;;
            "n") (( wygrane ++ ))
                 echo "$1: ${arr[$i]} - $comp" ;;
            *) echo "$1: ${arr[$i]} - $comp" ;;
        esac
        ;;
    "n")    case $comp in
            "p") (( wygrane++ ))
                 echo "$1: ${arr[$i]} - $comp" ;;
            "k") (( porażki++ ))
                 echo "$1: ${arr[$i]} - $comp" ;;
            *) echo "$1: ${arr[$i]} - $comp" ;;
        esac
        ;;
    *) echo "Uwaga: Bledny symbol!" ;;
    esac
done
echo "Koniec gry. Wygrane: $wygrane, przegrane: $porażki."
