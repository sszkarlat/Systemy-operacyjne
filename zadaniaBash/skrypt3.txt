#!/bin/bash

# Losowanie liczby
random_number=$((RANDOM % 10))

while true; do
    echo "Podaj swoją liczbę:"
    read user_number

    if [ "$user_number" -eq "$random_number" ]; then
        echo "Gratulacje, podana liczba jest taka sama jak wylosowana!"
        break
    elif [ "$user_number" -lt "$random_number" ]; then
        echo "Podana liczba jest mniejsza od wylosowanej. Spróbuj ponownie:"
    else
        echo "Podana liczba jest większa od wylosowanej. Spróbuj ponownie:"
    fi
done
