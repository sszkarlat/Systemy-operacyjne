RANGE=100
losowa=$RANDOM
let "losowa %= $RANGE"

liczba=
echo "Wprowadz liczbe 1-100"; read liczba

while [ $liczba -ne $losowa ]
do
    if [ $liczba -lt $losowa ]
    then
        echo "Za mala. Wprowadz liczbe 0-100"; read liczba

        echo "Za duza. Wprowadz liczbe 0-100"; read liczba
    fi
done
