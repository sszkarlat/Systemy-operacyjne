Zadanie napisane w bash
Mediana podanych liczb całkowitych
#!/bin/bash

mediana() {
        local numbers=($(echo "$1" | tr ' ' '\n' | sort -n))
        local len=${#numbers[@]}
        local middle=$((len/2))

        echo "${numbers[middle]}"
}


if [ "$#" -eq 0 ];then
        echo "Uzycie $0 <lista_liczb>"
        exit
elif [ $(( $# % 2)) -eq 0 ];then
        echo "Liczba liczb musi byc nieparzysta"
        exit
else
        data="$@"
        echo $(mediana "$data")
fi
